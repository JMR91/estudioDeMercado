\paragraph{Trayectorias Principal Alta estacionamiento}
	\begin{enumerate}
		\setlength\itemsep{0cm}
		\item \iconUser Preciona el botón 'Registrarse'
		\item \iconCloud Muestra la pantalla \ref{md:altaAdministrador}
		\item \iconCloud Solicita los datos para el registro. \ref{rn:altaAdministrador}
		\item \iconUser Ingresa los datos 
		\item \iconCloud Valida los datos \textit{\textbf{\nameref{parRegistroAdministrador:TrayectoriaA}}}
		\item \iconCloud Valida que no este regristrado \textit{\textbf{\nameref{parRegistroAdministrador:TrayectoriaB}}}
		\item \iconCloud Registra al usuario administrador 
	\end{enumerate}
